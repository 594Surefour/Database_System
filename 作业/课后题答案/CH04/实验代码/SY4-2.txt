题目2
（1）SELECT student.sno,sname,cname,grade FROM student INNER JOIN sc INNER JOIN course 
ON student.sno=sc.sno AND course.cno=sc.cno WHERE grade<60 and cname='MySQL数据库设计';
（2）SELECT student.sno,sname,cname,grade FROM student INNER JOIN sc INNER JOIN course 
ON student.sno=sc.sno AND course.cno=sc.cno WHERE sname='李四';
（3）SELECT student.sno,sname,count(cno),max(grade),min(grade),sum(grade),avg(grade)
FROM student left JOIN sc ON student.sno=sc.sno
GROUP BY student.sno;
（4）SELECT student.sno,sname,avg(grade)
FROM student INNER JOIN sc ON student.sno=sc.sno
GROUP BY student.sno
HAVING avg(grade)>70
ORDER BY avg(grade) DESC;
（5）SELECT student.sno,sname,cname,grade FROM student INNER JOIN sc INNER JOIN course 
ON student.sno=sc.sno AND course.cno=sc.cno WHERE cname='MySQL数据库设计';
